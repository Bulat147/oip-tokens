  // –ü–µ—Ä–µ–¥ —Ç–µ–º, –∫–∞–∫ –Ω–µ–π—Ä–æ–Ω—á–∏–∫ –≤—ã—Ä–∞—Å—Ç–µ—Ç –∏ —Å–º–æ–∂–µ—Ç –ø—Ä–∏–Ω–∏–º–∞—Ç—å —Ä–µ—à–µ–Ω–∏—è —Å–∞–º–æ—Å—Ç–æ—è—Ç–µ–ª—å–Ω–æ,
  // –º—ã –¥–æ–ª–∂–Ω—ã –æ–±—ä—è—Å–Ω–∏—Ç—å –µ–º—É, —á—Ç–æ —Ç–∞–∫–æ–µ —Ö–æ—Ä–æ—à–æ –∏ —á—Ç–æ —Ç–∞–∫–æ–µ –ø–ª–æ—Ö–æ —Å –ø–æ–º–æ—â—å—é
  // —Ç—Ä–µ–Ω–∏—Ä–æ–≤–æ—á–Ω—ã—Ö –ø—Ä–∏–º–µ—Ä–æ–≤.
  // xTrain -> [0, 1, 2, ...],
  // yTrain -> [32, 33.8, 35.6, ...]
  const xTrain = []
  const yTrain = []
  for (let x = 0; x < 100; x += 1) {
    const y = celsiusToFahrenheit(x)
    xTrain.push(x)
    yTrain.push(y)
  }

  // –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º –¢–ï–°–¢–û–í–´–ï –¥–∞–Ω–Ω—ã–µ.
  // –≠—Ç–∏ –¥–∞–Ω–Ω—ã–µ –±—É–¥—É—Ç –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å—Å—è –¥–ª—è –æ—Ü–µ–Ω–∫–∏ —Ç–æ–≥–æ, –Ω–∞—Å–∫–æ–ª—å–∫–æ —Ö–æ—Ä–æ—à–æ –º–æ–¥–µ–ª—å —Ä–∞–±–æ—Ç–∞–µ—Ç —Å –¥–∞–Ω–Ω—ã–º–∏,
  // –∫–æ—Ç–æ—Ä—ã—Ö –æ–Ω–∞ –Ω–µ –≤–∏–¥–µ–ª–∞ –≤ –ø—Ä–æ—Ü–µ—Å—Å–µ –æ–±—É—á–µ–Ω–∏—è. –ó–¥–µ—Å—å –º—ã –º–æ–∂–µ–º —É–≤–∏–¥–µ—Ç—å,
  // —á—Ç–æ –Ω–∞—à "—Ä–µ–±–µ–Ω–æ–∫" –≤—ã—Ä–æ—Å –∏ –º–æ–∂–µ—Ç –ø—Ä–∏–Ω–∏–º–∞—Ç—å —Ä–µ—à–µ–Ω–∏—è —Å–∞–º–æ—Å—Ç–æ—è—Ç–µ–ª—å–Ω–æ.
  // xTest -> [0.5, 1.5, 2.5, ...]
  // yTest -> [32.9, 34.7, 36.5, ...]
  const xTest = []
  const yTest = []
  // –ù–∞—á–∏–Ω–∞—è —Å 0.5 –∏ –∏—Å–ø–æ–ª—å–∑—É—è —Ç–∞–∫–æ–π –∂–µ —à–∞–≥ 1,
  // –∫–æ—Ç–æ—Ä—ã–π –º—ã –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–ª–∏ –¥–ª—è —Ç—Ä–µ–Ω–∏—Ä–æ–≤–æ—á–Ω–æ–≥–æ –Ω–∞–±–æ—Ä–∞,
  // –º—ã –æ–±–µ—Å–ø–µ—á–∏–≤–∞–µ–º —É–Ω–∏–∫–∞–ª—å–Ω–æ—Å—Ç—å –¥–∞–Ω–Ω—ã—Ö
  for (let x = 0.5; x < 100; x += 1) {
    const y = celsiusToFahrenheit(x)
    xTest.push(x)
    yTest.push(y)
  }

  return [xTrain, yTrain, xTest, yTest]
}

// –í—ã—á–∏—Å–ª—è–µ–º —Å—Ç–æ–∏–º–æ—Å—Ç—å (–æ—à–∏–±–∫—É) –º–µ–∂–¥—É –ø—Ä–∞–≤–∏–ª—å–Ω—ã–º –∑–Ω–∞—á–µ–Ω–∏–µ–º 'y' –∏
// 'prediction' (–ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–µ–º), —Å–¥–µ–ª–∞–Ω–Ω—ã–º –Ω–µ–π—Ä–æ–Ω—á–∏–∫–æ–º
function predictionCost(y, prediction) {
  // –≠—Ç–æ –ø—Ä–æ—Å—Ç–æ —Ä–∞–∑–Ω–∏—Ü–∞ –º–µ–∂–¥—É –¥–≤—É–º—è –∑–Ω–∞—á–µ–Ω–∏—è–º–∏.
  // –ß–µ–º –±–ª–∏–∂–µ –∑–Ω–∞—á–µ–Ω–∏—è –¥—Ä—É–≥ –∫ –¥—Ä—É–≥—É, —Ç–µ–º –º–µ–Ω—å—à–µ —Ä–∞–∑–Ω–∏—Ü–∞.
  // –ú—ã –∏—Å–ø–æ–ª—å–∑—É–µ–º –∑–¥–µ—Å—å —Å—Ç–µ–ø–µ–Ω—å 2 —Ç–æ–ª—å–∫–æ –¥–ª—è —Ç–æ–≥–æ, —á—Ç–æ–±—ã –∏–∑–±–∞–≤–∏—Ç—å—Å—è –æ—Ç –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã—Ö —á–∏—Å–µ–ª,
  // –ø–æ—ç—Ç–æ–º—É (1 - 2) ^ 2 = (2 - 1) ^ 2.
  // –†–µ–∑—É–ª—å—Ç–∞—Ç –¥–µ–ª–∏—Ç—Å—è –Ω–∞ 2 –ø—Ä–æ—Å—Ç–æ –¥–ª—è —É–ø—Ä–æ—â–µ–Ω–∏—è –¥–∞–ª—å–Ω–µ–π—à–µ–π —Ñ–æ—Ä–º—É–ª—ã –æ–±—Ä–∞—Ç–Ω–æ–≥–æ —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–∏—è (—Å–º. –Ω–∏–∂–µ)
  return (y - prediction) ** 2 / 2 // –Ω–∞–ø—Ä–∏–º–µ—Ä -> 235.6
}

// –ü—Ä—è–º–æ–µ —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–∏–µ.
// –≠—Ç–∞ —Ñ—É–Ω–∫—Ü–∏—è –±–µ—Ä–µ—Ç –≤—Å–µ –ø—Ä–∏–º–µ—Ä—ã –∏–∑ —Ç—Ä–µ–Ω–∏—Ä–æ–≤–æ—á–Ω—ã—Ö –Ω–∞–±–æ—Ä–æ–≤ xTrain –∏ yTrain
// –∏ –≤—ã—á–∏—Å–ª—è–µ—Ç –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è –º–æ–¥–µ–ª–∏ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –ø—Ä–∏–º–µ—Ä–∞ –∏–∑ xTrain.
// –ü–æ –ø—É—Ç–∏ –æ–Ω–∞ —Ç–∞–∫–∂–µ –≤—ã—á–∏—Å–ª—è–µ—Ç —Å—Ä–µ–¥–Ω—é—é —Å—Ç–æ–∏–º–æ—Å—Ç—å –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–π
function forwardPropagation(model, xTrain, yTrain) {
  const m = xTrain.length
  const predictions = []
  let cost = 0
  for (let i = 0; i < m; i += 1) {
    const prediction = model.predict(xTrain[i])
    cost += predictionCost(yTrain[i], prediction)
    predictions.push(prediction)
  }
  // –ù–∞—Å –∏–Ω—Ç–µ—Ä–µ—Å—É–µ—Ç —Å—Ä–µ–¥–Ω—è—è —Å—Ç–æ–∏–º–æ—Å—Ç—å
  cost /= m
  return [predictions, cost]
}

// –û–±—Ä–∞—Ç–Ω–æ–µ —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–∏–µ.
// –í —ç—Ç–æ–º –º–µ—Å—Ç–µ –º–∞—à–∏–Ω–Ω–æ–µ –æ–±—É—á–µ–Ω–∏–µ –≤—ã–≥–ª—è–¥–∏—Ç –∫–∞–∫ –º–∞–≥–∏—è.
// –ö–ª—é—á–µ–≤–æ–π –∫–æ–Ω—Ü–µ–ø—Ü–∏–µ–π –∑–¥–µ—Å—å —è–≤–ª—è–µ—Ç—Å—è –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–∞—è (derivative), –∫–æ—Ç–æ—Ä–∞—è –ø–æ–∫–∞–∑—ã–≤–∞–µ—Ç, –∫–∞–∫–æ–π —à–∞–≥ –Ω—É–∂–Ω–æ –ø—Ä–µ–¥–ø—Ä–∏–Ω—è—Ç—å, —á—Ç–æ–±—ã
// –ø—Ä–∏–±–ª–∏–∑–∏—Ç—å—Å—è –∫ –º–∏–Ω–∏–º—É–º—É —Ñ—É–Ω–∫—Ü–∏–∏ —Å—Ç–æ–∏–º–æ—Å—Ç–∏. –ü–æ–º–Ω–∏—Ç–µ, –Ω–∞—Ö–æ–∂–¥–µ–Ω–∏–µ –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ —Å—Ç–æ–∏–º–æ—Å—Ç–∏ -
// –∫–æ–Ω–µ—á–Ω–∞—è —Ü–µ–ª—å –ø—Ä–æ—Ü–µ—Å—Å–∞ –æ–±—É—á–µ–Ω–∏—è. –§—É–Ω–∫—Ü–∏—è —Å—Ç–æ–∏–º–æ—Å—Ç–∏ –≤—ã–≥–ª—è–¥–∏—Ç —Å–ª–µ–¥—É—é—â–∏–º –æ–±—Ä–∞–∑–æ–º:
// (y - prediction) ^ 2 * 1/2, –≥–¥–µ prediction = x * w + b.
function backwardPropagation(predictions, xTrain, yTrain) {
  const m = xTrain.length
  // –í –Ω–∞—á–∞–ª–µ –º—ã –Ω–µ –∑–Ω–∞–µ–º, –∫–∞–∫ –º–µ–Ω—è—Ç—å –ø–∞—Ä–∞–º–µ—Ç—Ä—ã 'w' –∏ 'b'.
  // –ü–æ—ç—Ç–æ–º—É —É—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º —à–∞–≥ –∏–∑–º–µ–Ω–µ–Ω–∏—è –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞ –≤ –∑–Ω–∞—á–µ–Ω–∏–µ 0
  let dW = 0
  let dB = 0
  for (let i = 0; i < m; i += 1) {
    // –≠—Ç–æ –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏–∏ —Å—Ç–æ–∏–º–æ—Å—Ç–∏ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞ 'w'.
    // –û–Ω–∞ –ø–æ–∫–∞–∑—ã–≤–∞–µ—Ç, –≤ –∫–∞–∫–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–∏ (–ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã–π/–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–π –∑–Ω–∞–∫ 'dW') –∏
    // –Ω–∞ —Å–∫–æ–ª—å–∫–æ (–∞–±—Å–æ–ª—é—Ç–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ 'dW') –ø–∞—Ä–∞–º–µ—Ç—Ä 'w' –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å –∏–∑–º–µ–Ω–µ–Ω
    dW += (yTrain[i] - predictions[i]) * xTrain[i]
    // –≠—Ç–æ –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏–∏ —Å—Ç–æ–∏–º–æ—Å—Ç–∏ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞ 'b'.
    // –û–Ω–∞ –ø–æ–∫–∞–∑—ã–≤–∞–µ—Ç, –≤ –∫–∞–∫–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–∏ (–∑–Ω–∞–∫ 'dB') –∏
    // –Ω–∞ —Å–∫–æ–ª—å–∫–æ (–∞–±—Å–æ–ª—é—Ç–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ 'dB') –ø–∞—Ä–∞–º–µ—Ç—Ä 'b' –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å –∏–∑–º–µ–Ω–µ–Ω
    dB += yTrain[i] - predictions[i]
  }
  // –ù–∞—Å –∏–Ω—Ç–µ—Ä–µ—Å—É—é—Ç —Å—Ä–µ–¥–Ω—è—è –¥–µ–ª—å—Ç–∞ –∫–∞–∂–¥–æ–≥–æ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞
  dW /= m
  dB /= m
  return [dW, dB]
}

// –û–±—É—á–∞–µ—Ç –º–æ–¥–µ–ª—å.
// –≠—Ç–æ "—É—á–∏—Ç–µ–ª—å" –Ω–∞—à–µ–≥–æ –Ω–µ–π—Ä–æ–Ω—á–∏–∫–∞:
// - –æ–Ω –ø—Ä–æ–≤–æ–¥–∏—Ç –Ω–µ–∫–æ—Ç–æ—Ä–æ–µ –≤—Ä–µ–º—è (epochs) —Å –Ω–∞—à–∏–º –≥–ª—É–ø—ã–º –Ω–µ–π—Ä–æ–Ω—á–∏–∫–æ–º –∏ –ø—ã—Ç–∞–µ—Ç—Å—è –µ–≥–æ —á–µ–º—É-—Ç–æ –Ω–∞—É—á–∏—Ç—å,
// - –æ–Ω –∏—Å–ø–æ–ª—å–∑—É–µ—Ç —Å–ø–µ—Ü–∏–∞–ª—å–Ω—ã–µ "–∫–Ω–∏–≥–∏" (–Ω–∞–±–æ—Ä—ã –¥–∞–Ω–Ω—ã—Ö xTrain –∏ yTrain) –¥–ª—è –æ–±—É—á–µ–Ω–∏—è,
// - –æ–Ω –∑–∞—Å—Ç–∞–≤–ª—è–µ—Ç —Ä–µ–±–µ–Ω–∫–∞ —É—á–∏—Ç—å—Å—è —É—Å–µ—Ä–¥–Ω–µ–µ (–±—ã—Å—Ç—Ä–µ–µ) —Å –ø–æ–º–æ—â—å—é –ø–∞—Ä–∞–º–µ—Ç—Ä–∞ –æ—Ü–µ–Ω–∫–∏ –æ–±—É—á–µ–Ω–∏—è 'alpha'
// (—á–µ–º —Å–∏–ª—å–Ω–µ–µ —Å—Ç–∏–º—É–ª, —Ç–µ–º –±—ã—Å—Ç—Ä–µ–µ –º–æ–¥–µ–ª—å —É—á–∏—Ç—Å—è, –Ω–æ –µ—Å–ª–∏ —É—á–∏—Ç–µ–ª—å –±—É–¥–µ—Ç –¥–∞–≤–∏—Ç—å —Å–ª–∏—à–∫–æ–º —Å–∏–ª—å–Ω–æ
// —É "—Ä–µ–±–µ–Ω–∫–∞" –º–æ–∂–µ—Ç —Å–ª—É—á–∏—Ç—å—Å—è –Ω–µ—Ä–≤–Ω—ã–π —Å—Ä—ã–≤, –∏ –±–æ–ª—å—à–µ –æ–Ω –Ω–µ —Å–º–æ–∂–µ—Ç —É—á–∏—Ç—å—Å—è)
function trainModel(model, epochs, alpha, xTrain, yTrain) {
  // –ò—Å—Ç–æ—Ä–∏—è –æ–±—É—á–µ–Ω–∏—è –º–æ–¥–µ–ª–∏.
  // –û–Ω–∞ –º–æ–∂–µ—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å —Ö–æ—Ä–æ—à–∏–µ –∏–ª–∏ –ø–ª–æ—Ö–∏–µ "–æ—Ü–µ–Ω–∫–∏" (—Å—Ç–æ–∏–º–æ—Å—Ç—å),
  // –ø–æ–ª—É—á–µ–Ω–Ω—ã–µ –≤ –ø—Ä–æ—Ü–µ—Å—Å–µ –æ–±—É—á–µ–Ω–∏—è
  const costHistory = []

  // –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º —ç–ø–æ—Ö–∏
  for (let i = 0; i < epochs; i += 1) {
    // –ü—Ä—è–º–æ–µ —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–∏–µ –¥–ª—è –≤—Å–µ—Ö —Ç—Ä–µ–Ω–∏—Ä–æ–≤–æ—á–Ω—ã—Ö –ø—Ä–∏–º–µ—Ä–æ–≤.
    // –°–æ—Ö—Ä–∞–Ω—è–µ–º —Å—Ç–æ–∏–º–æ—Å—Ç—å —Ç–µ–∫—É—â–µ–π –∏—Ç–µ—Ä–∞—Ü–∏–∏.
    // –≠—Ç–æ –ø–æ–º–æ–∂–µ—Ç –∞–Ω–∞–ª–∏–∑–∏—Ä–æ–≤–∞—Ç—å –æ–±—É—á–µ–Ω–∏–µ –º–æ–¥–µ–ª–∏
    const [predictions, cost] = forwardPropagation(model, xTrain, yTrain)
    costHistory.push(cost)

    // –û–±—Ä–∞—Ç–Ω–æ–µ —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–∏–µ. –£—á–∏–º—Å—è –Ω–∞ –æ—à–∏–±–∫–∞—Ö.
    // –≠—Ç–∞ —Ñ—É–Ω–∫—Ü–∏—è –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –Ω–µ–±–æ–ª—å—à–∏–µ –º–æ–¥–∏—Ñ–∏–∫–∞—Ü–∏–∏, –∫–æ—Ç–æ—Ä—ã–µ –Ω—É–∂–Ω–æ –ø—Ä–∏–º–µ–Ω–∏—Ç—å –∫ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º 'w' –∏ 'b',
    // —á—Ç–æ–±—ã —Å–¥–µ–ª–∞—Ç—å –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è –±–æ–ª–µ–µ —Ç–æ—á–Ω—ã–º–∏
    const [dW, dB] = backwardPropagation(predictions, xTrain, yTrain)

    // –ú–æ–¥–∏—Ñ–∏—Ü–∏—Ä—É–µ–º –ø–∞—Ä–∞–º–µ—Ç—Ä—ã –Ω–µ–π—Ä–æ–Ω—á–∏–∫–∞ –¥–ª—è –ø–æ–≤—ã—à–µ–Ω–∏—è —Ç–æ—á–Ω–æ—Å—Ç–∏ –µ–≥–æ –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–π
    nanoNeuron.w += alpha * dW
    nanoNeuron.b += alpha * dB
  }

  // –í–æ–∑–≤—Ä–∞—â–∞–µ–º –∏—Å—Ç–æ—Ä–∏—é –æ–±—É—á–µ–Ω–∏—è –¥–ª—è –∞–Ω–∞–ª–∏–∑–∞ –∏ –≤–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏–∏
  return costHistory
}

// –í –¥–∞–Ω–Ω—ã–π –º–æ–º–µ–Ω—Ç –Ω–µ–π—Ä–æ–Ω—á–∏–∫—É –Ω–µ–∏–∑–≤–µ—Å—Ç–Ω—ã –∑–Ω–∞—á–µ–Ω–∏—è –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ 'w' –∏ 'b'.
// –£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –∏—Ö –ø—Ä–æ–∏–∑–≤–æ–ª—å–Ω–æ
const w = Math.random() // –Ω–∞–ø—Ä–∏–º–µ—Ä -> 0.9492

const [xTrain, yTrain, xTest, yTest] = generateDataSets()

const epochs = 70000

export default function kMeans(data, k = 1) {
  if (!data) {
    throw new Error('–û—Ç—Å—É—Ç—Å—Ç–≤—É—é—Ç –¥–∞–Ω–Ω—ã–µ –¥–ª—è –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏')

  // –ö–ª–∞—Å—Å—ã –≤–µ–∫—Ç–æ—Ä–Ω—ã—Ö —Ç–æ—á–µ–∫ –¥–∞–Ω–Ω—ã—Ö.

  // –ò–Ω–¥–∏–∫–∞—Ç–æ—Ä –∏—Ç–µ—Ä–∞—Ü–∏–∏

  while (iterate) {

    // –í—ã—á–∏—Å–ª—è–µ–º –∏ —Å–æ—Ö—Ä–∞–Ω—è–µ–º —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –∫–∞–∂–¥–æ–π —Ç–æ—á–∫–∏ –æ—Ç –∫–∞–∂–¥–æ–≥–æ —Ü–µ–Ω—Ç—Ä–æ–∏–¥–∞

      }

      // –ü—Ä–∏—Å–≤–∞–∏–≤–∞–µ–º –º–µ—Ç–∫—É –±–ª–∏–∂–∞–π—à–µ–≥–æ –∫–ª–∞—Å—Ç–µ—Ä–∞ –∫–∞–∂–¥–æ–π —Ç–æ—á–∫–µ

        )
      }
    }
  }

}
